Margin Convention


var w = window.innerWidth,
    h = window.innerHeight,
    margin = 40

// select the .chart element and append an svg with attrs
var svg = d3.select('.chart')
    .append('svg') // append() returns a new selection
    .attr({
        width: w - margin * 2,
        height: h - margin * 2
    })

Good, but that needs some scale


var heightScale = d3.scale.linear()
    .domain([0, d3.max(data, function (d) { return d.value})])
    .range([0, svgHeight])

// ordinal scales for discrete data
var xScale = d3.scale.ordinal()
    .domain(data.map(function (d) { return d.category; }))
    .rangeRoundBands([0, svgWidth], 0.1);

Use scales for rect's attr

                           
.attr({
    x: function (d, i) {
        return xScale(d.category)
    },
    y: function (d, i) {
        return svgHeight - heightScale(d.value);
    },
    height: function (d) {
        return heightScale(d.value);
    },
